Integration

Two steps.
That's the entire integration.

1. Register
2. Build your callback
3. Verify a receipt
# Register your tool — one POST, you're done
curl -X POST https://zeemopay-production.up.railway.app/register \
  -H "Content-Type: application/json" \
  -d '{
    "wallet_address": "0xYOUR_WALLET",
    "tool_name":      "your-tool-name",
    "price_per_call": "0.10",
    "callback_url":   "https://your-server.com/your-tool",
    "timeout_seconds": 10
  }'

# Response
{
  "status":         "registered",
  "tool_name":      "your-tool-name",
  "price_per_call": "0.10",
  "pay_endpoint":   "/pay/your-tool-name"
}
# Your server receives this POST on every paid call
# X-Zeemo-Verified: true header confirms payment cleared

from fastapi import FastAPI, Request

app = FastAPI()

@app.post("/your-tool")
async def handle(request: Request):
    body = await request.json()
    user_input = body["input"]

    # Do your thing — LLM call, data lookup, whatever
    result = your_tool_logic(user_input)

    # Return anything JSON-serializable
    return {"result": result}

# That's it. Zeemo handles all the payment logic.
# Every transaction gets a tamper-evident receipt
curl https://zeemopay-production.up.railway.app/receipt/{transaction_id}

# Response
{
  "transaction_id":   "uuid",
  "tool_name":        "your-tool-name",
  "status":           "completed",
  "price_usdc":       0.10,
  "developer_cut":    0.095,
  "zeemo_cut":        0.005,
  "payout_tx_hash":   "0x...",
  "developer_wallet": "0x..."
}

# Verify payout_tx_hash on BaseScan — it's real on-chain proof
